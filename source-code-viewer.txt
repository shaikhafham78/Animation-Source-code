            function checkScaleAndSetOpacity() {
                document.querySelectorAll('.explode').forEach(explode => {
                    // Получаем текущую трансформацию элемента
                    const transform = window.getComputedStyle(explode).transform;
                    if (transform !== 'none') {
                        // Извлекаем значение scale из матрицы трансформации
                        const matrixValues = transform.match(/matrix.*\((.+)\)/)[1].split(', ');
                        const scaleX = parseFloat(matrixValues[0]);
                        // Проверяем значение scale и устанавливаем opacity
                        if (scaleX <= 0.1) {
                            explode.style.opacity = '0';
                        } else {
                            explode.style.opacity = '1';
                        }
                    }
                });
            }
            // Вызываем функцию для проверки при загрузке страницы и через определенные интервалы
